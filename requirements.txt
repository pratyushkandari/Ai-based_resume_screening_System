# requirements.txt
# =============================================================================
#  PURPOSE OF THIS FILE:
# =============================================================================
# The `requirements.txt` file lists all the Python libraries that your backend
# project depends on.
#
# Anyone setting up your project (including you, your teammates, or the teacher)
# can install all dependencies by simply running:
#
#     pip install -r requirements.txt
#
#  It ensures everyone uses the same versions of packages.
#  Used in FastAPI + PostgreSQL + ML projects like this resume screening system.
# =============================================================================


# -----------------------------------------------------------------------------
#  FASTAPI
# -----------------------------------------------------------------------------
# FastAPI is a modern, high-performance web framework for building REST APIs.
# It is the “core” of your backend — it defines the API endpoints (routes) like:
#   - /api/resumes/upload
#   - /api/jobs
#   - /api/jobs/{id}/summary
#
# Features:
#    Built-in data validation using Pydantic
#    Auto-generated Swagger docs (/docs)
#    Super fast (built on Starlette + uvicorn)
# -----------------------------------------------------------------------------
fastapi


# -----------------------------------------------------------------------------
#  UVICORN (Server)
# -----------------------------------------------------------------------------
# Uvicorn is a **lightning-fast ASGI web server** that actually runs your FastAPI app.
#
# Command to start your app:
#   uvicorn app.main:app --reload
#
# - “--reload” automatically restarts the server on code changes (for development)
# - This is what listens on port 8000 and serves API requests
# -----------------------------------------------------------------------------
uvicorn[standard]


# -----------------------------------------------------------------------------
#  SQLALCHEMY (ORM)
# -----------------------------------------------------------------------------
# SQLAlchemy is a Python **Object-Relational Mapper (ORM)**.
# It lets you work with your PostgreSQL database using Python classes instead of raw SQL.
#
# Example:
#   Instead of writing:  SELECT * FROM jobs;
#   You can do:          db.query(Job).all()
#
# - models.py defines database tables using SQLAlchemy classes.
# -----------------------------------------------------------------------------
sqlalchemy


# -----------------------------------------------------------------------------
#  PSYCOPG2-BINARY
# -----------------------------------------------------------------------------
# Psycopg2 is the **PostgreSQL database adapter** for Python.
# SQLAlchemy uses it under the hood to connect and execute queries.
#
# The “-binary” version includes precompiled native libraries — easier to install.
#
# Example connection string (from .env):
#   postgresql://resume_user1:Resume123@localhost:5432/resume_db1
# -----------------------------------------------------------------------------
psycopg2-binary


# -----------------------------------------------------------------------------
#  PYTHON-DOTENV
# -----------------------------------------------------------------------------
# Loads environment variables from the `.env` file (like DATABASE_URL, UPLOAD_DIR)
# into your Python process.
#
# Example:
#   from dotenv import load_dotenv
#   load_dotenv()
#
# This lets your app read secrets or settings without hardcoding them in code.
# -----------------------------------------------------------------------------
python-dotenv


# -----------------------------------------------------------------------------
# PYTHON-DOCX
# -----------------------------------------------------------------------------
# This library allows your app to **read .docx files** (Microsoft Word format).
# Used in ml_utils.py → `extract_text_docx(path)` to extract resume text.
#
# When a user uploads a resume, this library reads its paragraphs safely.
# -----------------------------------------------------------------------------
python-docx


# -----------------------------------------------------------------------------
# SENTENCE-TRANSFORMERS (Machine Learning)
# -----------------------------------------------------------------------------
# This is an NLP (Natural Language Processing) library that provides **pretrained
# embedding models** like “all-MiniLM-L6-v2”.
#
# It converts text (e.g., job description or resume) into numerical vectors
# for semantic similarity comparison.
#
# Used in:
#   - ml_utils.embed_text()
#   - ml_utils.compute_final_score()
#
#  If model download fails, ml_utils falls back to a deterministic pseudo vector.
# -----------------------------------------------------------------------------
sentence-transformers


# -----------------------------------------------------------------------------
# NUMPY
# -----------------------------------------------------------------------------
# NumPy is the fundamental library for **numerical computations** in Python.
# Used for:
#   - Handling embedding arrays (vectors)
#   - Computing cosine similarity efficiently
#
# Example:
#   cosine = np.dot(a, b) / (||a|| * ||b||)
# -----------------------------------------------------------------------------
numpy


# -----------------------------------------------------------------------------
#  PYTHON-MULTIPART
# -----------------------------------------------------------------------------
# Enables FastAPI to handle **multipart/form-data** — the format used when
# uploading files (like resumes) via HTML forms.
#
# Without this, FastAPI wouldn’t understand `UploadFile = File(...)`.
# -----------------------------------------------------------------------------
python-multipart


# -----------------------------------------------------------------------------
#  AIOFILES
# -----------------------------------------------------------------------------
# Provides asynchronous file I/O support for FastAPI.
# This allows efficient non-blocking file reads/writes — useful for serving files,
# like when downloading resumes via /api/resumes/{id}/download.
# -----------------------------------------------------------------------------
aiofiles


# =============================================================================
#  QUICK SUMMARY
# =============================================================================
# | Package Name          | Purpose / Function                                |
# |------------------------|--------------------------------------------------|
# | fastapi                | Web framework to define API endpoints            |
# | uvicorn[standard]      | ASGI web server to run the FastAPI app           |
# | sqlalchemy             | ORM for database communication                   |
# | psycopg2-binary        | PostgreSQL driver used by SQLAlchemy             |
# | python-dotenv          | Loads environment variables from .env            |
# | python-docx            | Reads .docx files for resume text extraction     |
# | sentence-transformers  | Converts text into semantic embeddings           |
# | numpy                  | Handles numerical vector operations              |
# | python-multipart       | Enables file uploads in FastAPI                  |
# | aiofiles               | Async file operations for better performance     |
# =============================================================================
